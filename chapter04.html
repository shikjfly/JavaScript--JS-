4.1 基本类型和引用类型的值

1、基本类型：Undefined、Null、Boolean、Number和String，这5种基本类型是按值访问的。。还有一种是引用类型，就是对象了

2、引用类型，可以添加属性和方法，但是基本类型，不行

3、复制基本类型的值，空间相互独立，互不伤害，但是复制引用类型的值那就大大的不一样了，复制之后，引用类型之间的值是互通的

4、函数中参数的值传递，从外面传递到里面来的过程，就是按值传递，类似复制过程

function addTen(num){
   mum += 10；
   return num;
}
var count = 20;
var result = addTen(count);
alert(count); // 20 没变化
alert(result); // 30

5、检测类型，typeof检测基本类型， instanceof检测引用类型是Object么、是Array么，是Function么。。

6、执行环境及作用域，全局环境对象是window对象，程序退出或浏览器关闭才会销毁这个全局对象

var color = "blue";
function changeColor(){
  if(color == "blue"){
    color = "red" ;
  } else {
    color = "blue";
  }
}
changeColor();
alert("Color is now " + color);

9、有var是局部，没var的声明是全局变量

function add(num1,num2){
   var sum = num1 +num2;
   return sum;
}
var result = add(10,20) ; // 30
alert(sum); // 局部变量sum不能再外部访问，报错

function add(num1,num2){
   sum = num1 +num2;
   return sum;
}
var result = add(10,20) ; // 30
alert(sum); // 30 没有var是全局变量，可以访问
